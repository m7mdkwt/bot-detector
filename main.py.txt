    "password": "YOUR_PASSWORD_HERE",
    "sslmode": "require"
}

def get_db():
    return psycopg2.connect(**DATABASE_CONFIG)

# 🧠 تخزين مؤقت للطلبات (RAM)
requests_log = {}

@app.get("/")
def home():
    return {"message": "Bot Detector Running 🤖"}

@app.post("/detect")
def detect(data: dict):

    ip = data.get("ip")
    ua = data.get("user_agent", "")
    path = data.get("path", "/")

    score = 0
    reasons = []

    # 🧠 1) User-Agent check
    if any(bot in ua.lower() for bot in ["python", "curl", "bot", "scraper"]):
        score += 50
        reasons.append("suspicious user-agent")

    # ⚡ 2) Speed check
    now = time.time()

    if ip not in requests_log:
        requests_log[ip] = []

    requests_log[ip].append(now)

    recent = [t for t in requests_log[ip] if now - t < 10]

    if len(recent) > 10:
        score += 40
        reasons.append("high request rate")

    # 🎯 تحديد النتيجة
    if score >= 70:
        result = "bot"
    elif score >= 40:
        result = "medium"
    else:
        result = "human"

    # 💾 حفظ في الداتابيس
    db = None
    cur = None

    try:
        db = get_db()
        cur = db.cursor()

        cur.execute("""
            INSERT INTO bot_logs (ip, user_agent, path, score, result)
            VALUES (%s, %s, %s, %s, %s)
        """, (ip, ua, path, score, result))

        db.commit()

    except Exception as e:
        print("DB ERROR:", e)

    finally:
        if cur:
            cur.close()
        if db:
            db.close()

    return {
        "result": result,
        "score": score,
        "reasons": reasons
    }
